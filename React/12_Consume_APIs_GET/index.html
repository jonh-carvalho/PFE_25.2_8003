const REST_COUNTRIES_API = 'https://restcountries.com/v3.1/all?fields=name,capital,population,region,subregion,area,flag,flags,languages,currencies,cca3';

function useCountries() {
  const [countries, setCountries] = useState([]);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState(null);

  // Fun√ß√£o para buscar pa√≠ses da API

      // Verificar se a requisi√ß√£o foi bem-sucedida

      }

      const data = await response.json();

      // Ordenar pa√≠ses por nome para melhor UX
      const sortedCountries = data.sort((a, b) => 
        a.name.common.localeCompare(b.name.common)
      );

      console.log(`‚úÖ ${sortedCountries.length} pa√≠ses carregados com sucesso!`);
      setCountries(sortedCountries);

    } catch (error) {
      console.error('‚ùå Erro ao buscar pa√≠ses:', error);
      setError(error.message);
    } finally {
      setIsLoading(false);
    }
  };

  // Carregar pa√≠ses ao montar o componente
  useEffect(() => {
    fetchCountries();
  }, []);

  // Fun√ß√£o para recarregar dados

  };

    countries, 
    isLoading, 
    error, 
    refetch 
  };
}

export default useCountries;

    1. Debounce para Busca:¶

    // src/hooks/useDebounce.js
import { useState, useEffect } from 'react';

function useDebounce(value, delay) {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => {
      clearTimeout(handler);
    };
  }, [value, delay]);

  return debouncedValue;
}

export default useDebounce;

    2. Cache Simples:¶

    // src/services/countryAPI.js
const CACHE_KEY = 'countries_cache';
const CACHE_DURATION = 5 * 60 * 1000; // 5 minutos

export const fetchCountriesWithCache = async () => {
  // Verificar cache
  const cached = localStorage.getItem(CACHE_KEY);
  if (cached) {
    const { data, timestamp } = JSON.parse(cached);
    if (Date.now() - timestamp < CACHE_DURATION) {
      console.log('üì¶ Dados carregados do cache');
      return data;
    }
  }

  // Buscar da API
  const response = await fetch(REST_COUNTRIES_API);
  if (!response.ok) {
    throw new Error(`Erro HTTP: ${response.status}`);
  }

  const data = await response.json();

  // Salvar no cache
  localStorage.setItem(CACHE_KEY, JSON.stringify({
    data,
    timestamp: Date.now()
  }));

  return data;
};
